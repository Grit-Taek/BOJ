#include <iostream>
#include <vector>
#include <queue>

using namespace std;

int c[8];
vector<int>v[8];

void bfs(int start)
{
    queue<int> q;
    q.push(start);
    c[start] = true;
    while(!q.empty()) {
        int x = q.front();
        q.pop();
        printf("%d", x);
        for (int i = 0; i < v[x].size(); i++) {
            int y = v[x][i];
            if (!c[y]) {
                q.push(y);
                c[y] = true;
            }
        }
    }
}
